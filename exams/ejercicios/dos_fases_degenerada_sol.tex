\begin{equation}
\begin{split}
\mbox{max. } z = 2x_1 + 3x_2\\
s.a.:\\
2x_1 + 3x_2 \leq 40\\
2x_1 + 2x_2 \geq 5\\
2x_2 \geq 5\\
x_1,\,\,x_2\geq 0\\
\end{split}
\end{equation}

\vspace{0.5cm}
