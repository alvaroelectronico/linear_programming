\begin{equation}
\begin{split}
\mbox{max. } z = 3x_1 + 5x_2\\
s.a.:\\
2x_1 + 4x_2 \leq  25\\
x_1 \leq 8\\
2x_2 \leq 10\\
x_1,\,\,x_2 \in  \mathcal{Z}^+
\end{split}
\end{equation}
